country_event = {
    id = louisiana.3
    hidden = yes

    fire_only_once = yes
    is_triggered_only = no

    trigger = {
        OR = {
            NOT = {
                STX = {
                    controls_state = 621
                    controls_state = 42
                }
            }
            has_global_flag = {
                flag = red_river_war_has_begun
                days > 180
            }
        }
    }

    immediate = {
        if = {
            limit = {
                NOT = {
                    STX = {
                        controls_state = 621
                        controls_state = 42
                    }
                }
            }
            SLA = {
                country_event = {
                    id = louisiana.4 #supreme victory
                }
            }
        }
        else_if = {
            limit = {
                NOT = {
                    STX = {
                        OR = {
                            controls_state = 621
                            controls_state = 42
                        }
                    }
                }
            }
            SLA = {
                country_event = {
                    id = louisiana.5 #victory
                }
            }
        }
        else_if = {
            limit = {
                STX = {
                    OR = {
                        controls_state = 152
                        controls_state = 236
                        controls_state = 240
                        controls_state = 235
                    }
                    OR = {
                        controls_state = 75
                        controls_state = 20
                    }
                }
            }
            STX = {
                country_event = {
                    id = louisiana.6 #loss
                }
            }
        }
        else = {
            SLA = {
                country_event = {
                    id = louisiana.7 #stalemate
                }
            }
        }
    }
}
